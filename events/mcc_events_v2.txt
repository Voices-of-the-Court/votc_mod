mcc_event_v2.1003 = {
	type = character_event
	theme = diplomacy_family_focus
	title = mcc_event.1002.title
	desc = mcc_event.1002.desc
	window = character_event

	immediate = {
		clear_variable_list = mcc_characters
		clear_variable_list = mcc_characters_all
		clear_variable_list = mcc_characters_all_in_location
		clear_variable_list = mcc_characters_selected
		every_pinned_character = {
			limit = { 
				is_alive = yes
				not = { any_player = { this = prev } }
				this.location = scope:actor.location
			}
			scope:actor = {
				add_to_variable_list = {
					name = mcc_characters
					target = prev
				}
			}
		}
		every_pinned_character = {
			limit = { 
				is_alive = yes
				not = { any_player = { this = prev } }
			}
			scope:actor = {
				add_to_variable_list = {
					name = mcc_characters_all
					target = prev
				}
			}
		}
		scope:actor.location = {
			every_character_in_location = {
				limit = { 
					is_alive = yes
					not = { any_player = { this = prev } }
				}
				scope:actor = {
					add_to_variable_list = {
						name = mcc_characters_all_in_location
						target = prev
					}
				}
			}
		}

	}
	
	widget = {
        gui = "mcc_select_characters_widget"
        container = "custom_widgets_container"
    }

	option ={
		name = votc_talk_change_mcc_selection
		trigger_event = mcc_event_v2.1003
	}
	option ={
		#trigger = {
		#	has_global_variable = talk_scene
		#}
		name = votc_talk_selected
		trigger_event = mcc_event_v2.8999
	}
	
	option ={
		name = mcc_event.1002.exit
	}
}

			id = mcc_event_v2.9998

		}
